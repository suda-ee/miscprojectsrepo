\begin{question}
    电路如下图所示，试求其中受控源的功率。($5'$)
    \begin{figure}[H]
    \hfill
    \input fig01
    \end{figure}
    \examvspace*{5cm}
    \begin{answer}
    列回路方程
    \[
    3I_1+2I_1=2
    \]
    解得~$I_1=-\unit{0.4}{\ampere}$ ($3'$)，受控源发出功率为~$3I_1
    \times 1 \times 2I_1=-\unit{0.96}{\watt}$ ($2'$)。
    \end{answer}
\end{question}

\begin{question}
    求下图所示电路中的电压~$U_\mathrm{o}$。($10'$)
    \begin{figure}[H]
    \hfill
    \input fig03
    \end{figure}
    \examvspace*{7cm}
    \begin{answer}
    应用节点法。取左下角节点为参考节点，右下边为节点~3，左上边为节点~1，
    右上为节点~2 ($2'$)。则有方程
    \[\begin{aligned}
    0.2U_\mathrm{n1}-0.1U_\mathrm{n2} &=-2 \\
    -0.1U_\mathrm{n1}+0.15U_\mathrm{n2}-0.05U_\mathrm{n3} &= 10 \\
    -0.05U_\mathrm{n2}+0.15U\mathrm{n3} &=-2
    \end{aligned}\] ($4'$)解得~$U_\mathrm{n1}=\unit{40}{\volt}$，
    $U_\mathrm{n2}=\unit{100}{\volt}$，$U_\mathrm{n3}=\unit{20}{\volt}$
    ($2'$)，因此
    ~$U_\mathrm{o}=U_\mathrm{n2}-U_\mathrm{n3}=\unit{80}{\volt}$ ($2'$)。
    \end{answer}
\end{question}

\begin{question}
    求下图所示电路中的~$U_3$。($10'$)
    \begin{figure}[H]
    \hfill
    \input fig04
    \end{figure}
    \examvspace*{7cm}
    \begin{answer}
    本电路只有一个回路，有~$I_1=-6/(6+4)=\unit{-0.6}{\ampere}$ ($5'$)，

\begin{question}
    求图示电路的戴维宁或诺顿等效电路。($10'$)
    \begin{figure}[H]
    \hfill
    \input fig05
    \end{figure}
    \examvspace*{8cm}
    \begin{answer}
    用节点电压法可求得开路电压~$U_\mathrm{OC}=\unit{24}{\volt}$ ($4'$)。
    同样可求得此电路短路电流为~$I_\mathrm{SC}=\unit{6}{\ampere}$ ($4'$)。故等效

\begin{question}

    \end{answer}

\begin{question}
    给出图中电路的结点电压方程和回路电流方程。($10'$)
    \begin{figure}[H]
    \hfill
    \input fig08
    \end{figure}
    \examvspace*{7cm}
    \begin{answer}
    结点电压方程为
    \[\left.\begin{aligned}
    3\dot{U}_\mathrm{n1}-2\dot{I}-\dot{U}_\mathrm{n3} &= \dot{U}_\mathrm{S}
    \\
    -\dot{U}_\mathrm{n1}-\jimag
    0.1\times 2\dot{I}+\dot{U}_\mathrm{n3} &= 0 \\
    \dot{I} &= (\dot{U}_\mathrm{n3}-\dot{U}_\mathrm{n1})/1
    \end{aligned}\right\}\]
    网孔方程为~($\dot{I}_1$~和~$\dot{I}_2$~分别为下边两个网孔电流，顺时
    针，上边网孔电流逆时针)
    \[\left.\begin{aligned}
    2 \dot{I}_1 + \dot{I} &= \dot{U}_\mathrm{S} - 2 \dot{I} \\
    -\jimag 10 \dot{I} &= 2 \dot{I} \\
    \dot{I}_1 - \jimag 10 \dot{I}_2 +(2-\jimag 10)\dot{I} &= 0
    \end{aligned}\right\}\]
    \end{answer}
\end{question}

\begin{question}
    求下图中电流源发出的复功率。($10'$)
    \begin{figure}[H]
    \hfill
    \input fig09
    \end{figure}
    \examvspace*{7cm}
    \begin{answer}
    各列一个~KCL~和~KVL~方程可解得
    ~$\dot{I}_1=5\sqrt{2}\Varangle{-45\degree}$，

    \end{answer}
\end{question}

\begin{question}
    图示电路中，开关~S~动作前电路已达稳定状态，原合在位置~1，$t=0$~时
    ~S~由位置~1~合向位置~2，求
    ~$t\geq 0$~时电感电流~$i_L$~和电压~$u_L$~的响应。($15'$)
    \begin{figure}[H]
    \hfill
    \input fig06
    \end{figure}
    \examvspace*{10cm}
    \begin{answer}
    先求~$i_L(0_-)$，解得~$i_L(0_-)=\unit{5}{\ampere}$ ($3'$)。则
    ~$i_L(0_+)=i_L(0_-)$，容易看出~$i_L(\infty)=0$ ($1'$)。利用外加
    电源法可求得~$R_\mathrm{eq}=\unit{12}{\ohm}$ ($3'$)，因此~$\tau
    =L/R_\mathrm{eq}=\unit{1/4}{\second}$ ($3'$)。套用三要素法可得
    \[\begin{aligned}
    i_L(t)&=i_L(\infty)+[i_L(0_+)-i_L(\infty)]\mathrm{e}^{-t/\tau} \\
    &=5\mathrm{e}^{-4t}\unit{}{\ampere}\quad\quad (3')
    \end{aligned}\]
    $u_L(t)=-60\mathrm{e}^{-4t}\unit{}{\volt}$ ($2'$)。
    \end{answer}
\end{question}
