\begin{question} % 01
    电路如下图所示，试求其中受控源的功率。($5'$)
    \begin{figure}[H]
    \hfill\includegraphics{fig01}
    \end{figure}
    \examvspace*{3cm}
    \begin{answer}
    列回路方程
    \[
    2 \times 0.5 + 2U-U=0
    \]
    解得~$U=-\unit{1}{\volt}$ ($3'$)，受控源吸收功率为~$2U \times 0.5=-\unit{1}{\watt}$ ($2'$)。
    \end{answer}
\end{question}

\begin{question} %02
    求下图所示电路的电压~$U_\mathrm{cb}$。($8'$)
    \begin{figure}[H]
    \hfill\includegraphics{fig02}
    \end{figure}
    \examvspace*{4cm}
    \begin{answer}
    $U_1 = 2 \times 5 = \unit{10}{\volt}$，$20 \times 0.05 U_1 +
    U_\mathrm{cb} + 3 = 0$，$\therefore U_\mathrm{cb} = -
    \unit{7}{\volt}$。
    \end{answer}
\end{question}

\begin{question} % 03
    求下图所示电路中的电压比~$U_\mathrm{o} / U_\mathrm{s}$。($12'$)
    \begin{figure}[H]
    \hfill\includegraphics{fig03}
    \end{figure}
    \examvspace*{7cm}
    \begin{answer}
    \[\begin{split}
    U_\mathrm{o} &= 3 U_\mathrm{s} \\
    (U_\mathrm{s} / 2 - U_3) \times 1 &= U_\mathrm{s} + U_\mathrm{o} \\
    2 U_3 + U_\mathrm{o} &= 5 U_3
    \end{split}\]
    $\therefore U_\mathrm{s} = 10 U_3$，$U_\mathrm{o} / U_\mathrm{s} =
    0.3$。
    \end{answer}
\end{question}

\begin{question} % 05
    图示电路中，当电流源~$i_\mathrm{s1}$~和电压源~$u_\mathrm{s1}$~反向时
    ~($u_\mathrm{s2}$~不变)，电压~$u_\mathrm{ab}$~是原来~0.5~倍；当
    ~$i_\mathrm{s1}$~和~$u_\mathrm{s2}$~反向时~($u_\mathrm{s1}$~不变)，
电压~$u_\mathrm{ab}$~是原来的~0.3~倍。问：仅~$i_\mathrm{s1}$~反向
~($u_\mathrm{s1}$、$u_\mathrm{s2}$~均不变)~时，电压~$u_\mathrm{ab}$~应
为原来的几倍？($10'$)
    \begin{figure}[H]
    \hfill\includegraphics{fig05}
    \end{figure}
    \examvspace*{4cm}
    \begin{answer}
    \[ \left \{
    \begin{aligned}
    u_\mathrm{ab} &= k_1 u_\mathrm{s1} + k_2 u_\mathrm{s2} + k_3 i_\mathrm{s1} \\
    0.5 u_\mathrm{ab} &= - k_1 u_\mathrm{s1} + k_2 u_\mathrm{s2} - k_3 i_\mathrm{s1} \\
    0.3 u_\mathrm{ab} &= k_1 u_\mathrm{s1} - k_2 u_\mathrm{s2} - k_3 i_\mathrm{s1} \\
    x u_\mathrm{ab} &= k_1 u_\mathrm{s1} + k_2 u_\mathrm{s2} - k_3 i_\mathrm{s1} \\
    \end{aligned}
    \right .
    \]
    解得~$x = 1.8$。
    \end{answer}
\end{question}

\begin{question} % 06
    求下图中的电流~$I_\mathrm{x}$。($12'$)
    \begin{figure}[H]
    \hfill\includegraphics{fig06}
    \end{figure}
    \examvspace*{5cm}
    \begin{answer}
    取回路为网孔，顺时针方向，则
    \[\left \{
    \begin{aligned}
    I_{l1} &= \unit{5}{\ampere} \\
    10 I_{l2} - 10 I_{l3} &= -10 I_\mathrm{x} + 50 \\
    -20 I_{l1} - 10 I_{l2} + 40 I_{l3} &= -30 \\
    I_\mathrm{x} &= I_{l3}
    \end{aligned}
    \right . \]
    解得~$I_\mathrm{x} = \unit{3}{\ampere}$。
    \end{answer}
\end{question}

\begin{question} % 07
    求图中电路在~ab~端口的戴维宁或诺顿等效电路。($10'$)
    \begin{figure}[H]
        \hfill
        \includegraphics{fig07}
    \end{figure}
    \examvspace*{7cm}
    \begin{answer}
    内部独立源全部置零后，成为一简单串联电路，容易求得等效内阻为：
    $R_{\rm eq}=\unit{10}{\ohm}$(5~分)；更容易解得开路电压~$u_{\rm
    OC}=10-1\times 5=\unit{5}{\volt}$(5~分)。
    \end{answer}
\end{question}

\begin{question} % 08
    求下图电路中~$Z_\mathrm{L}$~能够获得的最大功率。($10'$)
    \begin{figure}[H]
    \hfill\includegraphics{fig08}
    \end{figure}
    \examvspace*{11cm}
    \begin{answer}
    根据最大功率传输定理可以求得~$Z_\mathrm{L}$~获得最大功率的条件为
    ~$Z_\mathrm{L}=Z_\mathrm{eq}^*=\unit{10\sqrt{2}\Varangle{45\degree}}{\ohm}$ ($3'$)。
    $Z_\mathrm{L}$~开路时用节点电压法易求得开路电压满足
    ~$(1/20-\frac{1}{\jimag 10}+\frac{1}{\jimag 20})\dot{U}_\mathrm{OC}=5\Varangle{0\degree}+100\Varangle{90\degree}/20$，解得~$\dot{U}_\mathrm{OC}=100\Varangle{0\degree}\unit{}{\volt}$
    ($4'$)。$Z_\mathrm{L}$~可获得的最大功率为
    ~$P_\mathrm{Lmax}=\frac{U_\mathrm{OC}^2}{4R_\mathrm{eq}}=\unit{250}{\watt}$
    ($3'$)。
    \end{answer}
\end{question}

\begin{question} %09
图示电路中~$t=0$~时开关~$S_1$~打开，$S_2$~闭合，在开关动作前，电路已达
稳定。试求~$t\geq 0$~时的~$u_L(t)$、$i_L(t)$。($10'$)
    \begin{figure}[H]
        \hfill
        \includegraphics{fig09}
    \end{figure}
    \examvspace*{15cm}
    \begin{answer}
    先求~$0_-$~时刻的~$i_L$。(以下图略
    )~此时电路稳定，电感相当于短路，$i_L(0_-)=10/1=\unit{10}{\ampere}$，开关动作后，由于换路时电感电流
    不跃变，故~$i_L(0_+)=i_L(0_-)=\unit{10}{\ampere}$。当~$t=\infty$~时
    ，电感相当于短路，可解得~$i_(\infty)=\unit{3}{\ampere}$。$R_{\rm
    eq}=6\parallel 3=\unit{2}{\ohm}$，$\tau=L/R_{\rm
    eq}=\unit{0.1}{\second}$，套用三要素法公式有
    \[\begin{aligned}
    i_L(t)&=i_L(\infty)+[i_L(0_+)-i_L(\infty)]{\rm e}^{-t/\tau}\\
    &=3+7{\rm e}^{-10t}\unit{}{\ampere}
    \end{aligned}\]
    \[
    u_L(t)=L\frac{{\rm d}i_L(t)}{{\rm d}t}=-14{\rm
    e}^{-10t}\unit{}{\volt}
    \]
    (每式~3~分)
    \end{answer}
\end{question}

\begin{question} % 10
    列出下图电路的结点电压方程。($10'$)
    \begin{figure}[H]
    \hfill\includegraphics{fig10}
    \end{figure}
    \examvspace*{5cm}
    \begin{answer}
    \[ \left \{
    \begin{aligned}
    \dot{U}_\mathrm{n1} &= \dot{U}_\mathrm{s} \\
    -\dot{U}_\mathrm{n1} + (1 + 1 + \frac{1}{1 + \jimag 8}) \dot{U}_\mathrm{n2} -
        \dot{U}_\mathrm{n3} &= 0 \\
    \jimag 8 \dot{U}_\mathrm{n1} - \dot{U}_\mathrm{n2} + (1 +
        1 + \jimag 8) \dot{U}_\mathrm{n3} &=
          \dot{I}_\mathrm{s}
    \end{aligned}
    \right .
    \]
    \end{answer}
\end{question}
