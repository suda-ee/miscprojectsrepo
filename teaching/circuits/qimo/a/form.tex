\begin{question} % 01

    \hfill\includegraphics{fig01}
    \end{figure}

    \begin{answer}
    $P_{\unit{2}{\ampere}} = 2(5 \times 2 - 3) =
    \unit{140}{\watt}$~（放出），$P_{\unit{3}{\volt}} = 3 (2 - 0.5) =
    \unit{2.5}{\watt}$~(放出)。
    \end{answer}

\begin{question} % 02
    电路如下图所示，试求~$I_1$~与~$U_{\mathrm{ab}}$。($10'$)
    \begin{figure}[H]
    \hfill\includegraphics{fig02}
    \end{figure}
    \examvspace*{7cm}
    \begin{answer}
    $I = 10 / 5 = \unit{2}{\ampere}$，$I = 0.9 I_1$，$I_1 = I / 9 =
    \unit{20 / 9}{\ampere}$；$U_{\mathrm{ab}} = (I_1 - 0.9 I_1) \times 4 =
    0.4 I_1 = \unit{8 / 9}{\volt}$。
    \end{answer}
\end{question}

\begin{question} % 05
    已知题图所示电路中的网络~N~是由线性电阻组成。当~$i_\mathrm{s} =
    \unit{1}{\ampere}$，$u_\mathrm{s} = \unit{2}{\volt}$~时，$i =
    \unit{5}{\ampere}$；当~$i_\mathrm{s} = -\unit{2}{\ampere}$，
    $u_\mathrm{s} = \unit{4}{\volt}$~时，$u = \unit{24}{\volt}$。试求当
    ~$i_\mathrm{s} = \unit{2}{\ampere}$，$u_\mathrm{s} =
    \unit{6}{\volt}$~时的电压~$u$。 ($10'$)
    \begin{figure}[H]
    \hfill\includegraphics{fig05}
    \end{figure}
    \examvspace*{3cm}
    \begin{answer}
    设~$u = k_1 i_s + k_2 u_s$，代入已知条件得~$k_1 = -17$，$k_2 = 1$，
    最后得~$u = -\unit{28}{\volt}$。
    \end{answer}
\end{question}

\begin{question} %06
    求图中的~$U_x$。 ($10'$)
    \begin{figure}[H]
    \hfill\includegraphics{fig06}
    \end{figure}
    \examvspace*{5cm}
    \begin{answer}
    取回路为网孔，网孔电流选顺时针方向
    \[ \left\{ \begin{aligned}
        I_{l1} &= \unit{0.15}{\ampere} \\
        I_{l3} &= 0.05 U_x \\
        -24 I_{l1} + 56 I_{l2} - 32 I_{l3} &= 20 \\
        U_x &= 32 (I_{l1} - I_{l3})
    \end{aligned} \right.
    \]
    解得~$U_x = \unit{1.8}{\volt}$。
    \end{answer}
\end{question}

\begin{question} % 08
    已知~$\dot{I}_{\mathrm{S}} = 6 \Varangle{0\degree}\unit{}{\ampere}$，为使
    ~$Z_\mathrm{L}$~获得最大功率，求负载阻抗~$Z_\mathrm{L}$~和吸收的功率
    ~$P_\mathrm{Lmax}$。($10'$)
    \begin{figure}[H]
    \hfill\includegraphics{fig08}
    \end{figure}
    \examvspace*{8cm}
    \begin{answer}
    $Z_\mathrm{i} = 6 + 8\jimag \unit{}{\ampere}$，$Z_\mathrm{L} = 6 -
    8\jimag \unit{}{\ohm}$。$\dot{I}_\mathrm{SC} = -\jimag 5 / (15 -
    \jimag 5) \times 6 \Varangle{0\degree} = 6
    \Varangle{-71.6\degree} \unit{}{\ampere}$，$P_\mathrm{Lmax} = 0.25 R
    I^2_\mathrm{SC} = \unit{54}{\watt}$。
    \end{answer}
\end{question}

\begin{question} % 09
    图示电路中，开关~S~动作前电路已达稳定状态，原合在位置~1，$t=0$~时
    ~S~由位置~1~合向位置~2，求
    ~$t\geq 0$~时电感电流~$i_L$~和电压~$u_L$~的响应。($15'$)
    \begin{figure}[H]
    \hfill\includegraphics{fig09}
    \end{figure}
    \examvspace*{14cm}
    \begin{answer}
    先求~$i_L(0_-)$，解得~$i_L(0_-)=\unit{5}{\ampere}$ ($3'$)。则
    ~$i_L(0_+)=i_L(0_-)$，容易看出~$i_L(\infty)=0$ ($1'$)。利用外加
    电源法可求得~$R_\mathrm{eq}=\unit{12}{\ohm}$ ($3'$)，因此~$\tau
    =L/R_\mathrm{eq}=\unit{1/4}{\second}$ ($3'$)。套用三要素法可得
    \[\begin{split}
    i_L(t)&=i_L(\infty)+[i_L(0_+)-i_L(\infty)]\mathrm{e}^{-t/\tau} \\
    &=5\mathrm{e}^{-4t}\unit{}{\ampere}\quad\quad (3')
    \end{split}\]
    $u_L(t)=-60\mathrm{e}^{-4t}\unit{}{\volt}$ ($2'$)。
    \end{answer}
\end{question}

    \end{aligned}\right\}\]
    \end{answer}
\end{question}
