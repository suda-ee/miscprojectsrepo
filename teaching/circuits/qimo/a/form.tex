\end{question}

\begin{shortanswer}[title={\begin{description}\item[二、]计算题~(每

\begin{question}
    已知题图所示电路中的网络~N~是由线性电阻组成。当
    ~$i_\mathrm{S}=\unit{1}{A}$，$u_\mathrm{S}=\unit{2}{V}$~时，
    $i=\unit{5}{A}$；当~$i_\mathrm{S}=\unit{-2}{A}$，
    $u_\mathrm{S}=\unit{4}{V}$~时，$u=\unit{24}{V}$。试求当
    ~$i_\mathrm{S}=\unit{2}{A}$，$u_\mathrm{S}=\unit{6}{V}$~时的电压~$u$。
    \examvspace{12pt}\\
    \includegraphics{fig4.mps}
    \examvspace*{4cm}
    \begin{answer}
    设：$u=k_1 i_\mathrm{S}+k_2 u_\mathrm{S}$ ($3'$)，代入已知条件得
    ~$k_1=-17$，$k_2=1$ ($4'$)。最后得~$u=\unit{-28}{\volt}$ ($3'$)。
    \end{answer}
\end{question}
\begin{question}
    求图示电路中的受控源两端的电压~$U$。
    \begin{figure}[H]
    \hfill\includegraphics{fig5.mps}
    \end{figure}
    \examvspace*{6cm}
    \begin{answer}
    取上边基本回路为~1~逆时针，下边两网孔分别为~2、3，顺时针，
    $I_{L1}=10I$；$I_{L2}=\unit{3}{\ampere}$。列回路~III~方程为
    ~$(1/2+1/3)I_{L3}-1/2I_{L2}+1/3I_{L1}=-5$，辅助方程~$3-I_{L3}=I$
    ($6'$)。得
    ~$I=\unit{-2.4}{\ampere}$，受控源两端电压
    ~$U=5-1/2I=\unit{6.2}{\volt}$ ($4'$)。
    \end{answer}
\end{question}
\begin{question}
    图示电路中开关~$S$~打开前已处于稳态。$t=0$~开关打开，求~$t>0$~时的
    ~$i_C(t)$。
    \examvspace{12pt}\\
    \includegraphics{fig6.mps}
    \examvspace*{12cm}
    \begin{answer}
    $u_C(0_+)=u_C(0_-)=\unit{6}{\volt}$，$u_C(\infty)=\unit{12}{\volt}$
    ($4'$)，$R_0=\unit{2}{\kilo\ohm}$，$\tau=R_0C=\unit{0.04}{\second}$
    ($2'$)，
    $u_C(t)=u_C(\infty)+[u_C(0_+)-u_C(\infty)]\mathrm{e}^{-t/\tau}=12-6\mathrm{e}^{-25t}\unit{}{\volt}$
    ($2'$)，
    $i_C=C\frac{\mathrm{d}u_C}{\mathrm{d}t}=3\mathrm{e}^{-25t}\unit{}{\ampere}$
    ($2'$)。
    \end{answer}
\end{question}

\end{question}

\begin{shortanswer}[title={\begin{description}\item[三、]计算分析题~(每
题~13~分，共~26~分)\end{description}}]
\begin{question}
    图示电路中~$t=0$~时开关~$S_1$~打开，$S_2$~闭合，在开关动作前，电路已达
    稳定。试求~$t\geq 0$~时的~$u_L(t)$、$i_L(t)$。
    \begin{figure}[H]
    \hfill\includegraphics{fig9}
    \end{figure}
    \examvspace*{16cm}
    \begin{answer}
    先求~$0_-$~时刻的~$i_L$。(以下图略
    )~此时电路稳定，电感相当于短路，$i_L(0_-)=10/1=\unit{10}{\ampere}$，开关动作后，由于换路时电感电流
    不跃变，故~$i_L(0_+)=i_L(0_-)=\unit{10}{\ampere}$。当~$t=\infty$~时
    ，电感相当于短路，可解得~$i_L(\infty)=\unit{3}{\ampere}$。$R_{\rm
    eq}=6\parallel 3=\unit{2}{\ohm}$，$\tau=L/R_{\rm
    eq}=\unit{0.1}{\second}$，套用三要素法公式有
    \[\begin{aligned}
    i_L(t)&=i_L(\infty)+[i_L(0_+)-i_L(\infty)]{\rm e}^{-t/\tau}\\
    &=3+7{\rm e}^{-10t}\unit{}{\ampere}
    \end{aligned}\]
    \[
    u_L(t)=L\frac{{\rm d}i_L(t)}{{\rm d}t}=-14{\rm
    e}^{-10t}\unit{}{\volt}
    \]
    (每式~$3'$，数值正确~$1'$)
    \end{answer}
\end{question}
\begin{question}
    已知附图电路中的电压源为正弦量，$L=\unit{1}{\milli\henry}$，
    $R_0=\unit{1}{\kilo\ohm}$，$Z=(3+\mathrm{j}5)\unit{}{\ohm}$。试求：
    (1)~当~$\dot{I}_0=0$~时，$C$~值为多少？(2)~当条件~(1)~满足时，试证明
    输入阻抗为~$R_0$。
    \begin{figure}[H]
    \hfill\includegraphics{fig10}
    \end{figure}
    \examvspace*{8cm}
    \begin{answer}
    根据分压原理，$\dot{I}_0=0$~时有：$R_0\cdot \mathrm{j}\omega
    C=\mathrm{j}\omega L/R_0$，解得：$C=\unit{1}{\nano\farad}$ ($7'$)。
    此时~$Z$~支路相当于开路，输入阻抗可表示为：$(R_0+1/\mathrm{j}\omega
    C)\parallel(\mathrm{j}\omega L+R_0)$，化简得：$R_0$ ($6'$)。
    \end{answer}
\end{question}
