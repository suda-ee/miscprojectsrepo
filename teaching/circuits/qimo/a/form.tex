\begin{question}
    电路如下图所示，试求其中受控源的功率。($5'$)
    \begin{figure}[H]
    \hfill\includegraphics{fig01}
    \end{figure}
    \examvspace*{5cm}
    \begin{answer}
    列回路方程
    \[
    2 \times 0.5 + 2U-U=0
    \]
    解得~$U=-\unit{1}{\volt}$ ($3'$)，受控源吸收功率为~$2U \times 0.5=-\unit{1}{\watt}$ ($2'$)。
    \end{answer}
\end{question}

\begin{question}
    求下图所示电路的电流~$I$。($10'$)
    \begin{figure}[H]
    \hfill\includegraphics{fig02}
    \end{figure}
    \examvspace*{7cm}
    \begin{answer}
    利用电源等效变换可求得~$I=\unit{0.125}{\ampere}$ (每步~$2'$)。
    \end{answer}
\end{question}

\begin{question}
    求下图所示电路中的~$U$~和~$I$。($10'$)
    \begin{figure}[H]
    \hfill\includegraphics{fig03.mps}
    \end{figure}
    \examvspace*{7cm}
    \begin{answer}
    应用节点法。取最上面节点为参考节点，右边为节点~1，左边为节点~2，
    中间为节点~3 ($2'$)。则~$U_\mathrm{n1}=\unit{100}{\volt}$，
    $U_\mathrm{n2}=100+110=\unit{210}{\volt}$ ($2'$)，有方程
    \[
    -0.5U_\mathrm{n1}-0.5U_\mathrm{n2}+U_\mathrm{n3}=20
    \] ($4'$)解得~$U_\mathrm{n3}=\unit{175}{\volt}$，$U=U_\mathrm{n3}+1\times
    20=\unit{195}{\volt}$，$I=-(U_\mathrm{n2}-90)/1=\unit{-120}{\ampere}$ ($2'$)。
    \end{answer}
\end{question}

\begin{question}
    求下图所示电路中的~$U_3$。($10'$)
    \begin{figure}[H]
    \hfill\includegraphics{fig04}
    \end{figure}
    \examvspace*{7cm}
    \begin{answer}
    用叠加定理，当电压源单独作用时有~$I_1'=I_2'=\unit{1}{\ampere}$ ($2'$)，
    $U_3=-10I_1'+4I_2'=\unit{-6}{\volt}$ ($3'$)。当电流源单独作用时有
    ~$I_1''=\unit{-1.6}{\ampere}$，$I_2''=4+I_1''=\unit{2.4}{\ampere}$ ($2'$)，
    $U_3''=-10I_1''+4I_2''=\unit{25.6}{\volt}$ ($2'$)。所以
    ~$U_3=U_3'+U_3''=\unit{19.6}{\volt}$ ($1'$)。
    \end{answer}
\end{question}

\begin{question}
    求图示电路的戴维宁或诺顿等效电路。($10'$)
    \begin{figure}[H]
    \hfill\includegraphics{fig05}
    \end{figure}
    \examvspace*{8cm}
    \begin{answer}
    开路电压~$U_\mathrm{OC}=2I_1-2I_1=0$ ($3'$)。用外加电源法求输入电阻
    ，加一电压源~$u$ ($1'$)，可解得伏安关系为~$u=7i$ ($4'$)，故等效电阻为~$R_\mathrm{eq}=u/i=\unit{7}{\ohm}$ ($2'$)。
    \end{answer}
\end{question}

\begin{question}
    已知下图电路中~$I_1=I_2=\unit{10}{\ampere}$，求~$\dot{I}$~和
    ~$\dot{U}_\mathrm{S}$。(设~$\dot{I}_1$~的初相角为~$0\degree$) ($10'$)
    \begin{figure}[H]
    \hfill\includegraphics{fig07}
    \end{figure}
    \examvspace*{8cm}
    \begin{answer}
    设~$\dot{I}_1=10\Varangle{0\degree}$ ($2'$)，则
    $\dot{I}_2=10\Varangle{90\degree}$ ($2'$)，于是
    ~$\dot{I}=\dot{I}_1+\dot{I}_2=10\sqrt{2}\Varangle{45\degree}$ ($3'$)。
    $\dot{U}_\mathrm{S}=\mathrm{j}10\dot{I}+10\dot{I}_1=\jimag
    \unit{100}{\volt}$ ($3'$)。
    \end{answer}
\end{question}

\begin{question}
    给出图中电路的结点电压方程和回路电流方程。($10'$)
    \begin{figure}[H]
    \hfill\includegraphics{fig08}
    \end{figure}
    \examvspace*{7cm}
    \begin{answer}
    结点电压方程为
    \[\left.\begin{aligned}
    -\frac{1}{\jimag 5}\dot{U}_{\rm S}+\dot{U}_\mathrm{n2}-\dot{U}_\mathrm{n3}&=0\\
    -\dot{U}_\mathrm{n2}+\dot{U}_\mathrm{n3}&=-g\dot{U}_{\rm o}\\
    \dot{U}_{\rm S}-\dot{U}_\mathrm{n3}&=\dot{U}_{\rm o}
    \end{aligned}\right\}\]
    网孔方程为
    \[\left.\begin{aligned}
    \jimag 5g\dot{U}_{\rm o}&=\dot{U}_{\rm S}\\
    \jimag 5\dot{I}_2+g\dot{U}_{\rm o}&=\dot{U}_{\rm o}
    \end{aligned}\right\}\]
    \end{answer}
\end{question}

\begin{question}
    求下图电路中~$Z_\mathrm{L}$~能够获得的最大功率。($10'$)
    \begin{figure}[H]
    \hfill\includegraphics{fig09}
    \end{figure}
    \examvspace*{8cm}
    \begin{answer}
    根据最大功率传输定理可以求得~$Z_\mathrm{L}$~获得最大功率的条件为
    ~$Z_\mathrm{L}=Z_\mathrm{eq}^*=\unit{20}{\ohm}$ ($3'$)。
    $Z_\mathrm{L}$~开路时用节点电压法易求得开路电压满足
    ~$(1/20)\dot{U}_\mathrm{OC}=5\Varangle{0\degree}+100\Varangle{90\degree}/20$，解得~$\dot{U}_\mathrm{OC}=100\sqrt{2}\Varangle{45\degree}\unit{}{\volt}$
    ($4'$)。$Z_\mathrm{L}$~可获得的最大功率为
    ~$P_\mathrm{Lmax}=\frac{U_\mathrm{OC}^2}{4R_\mathrm{eq}}=\unit{250}{\watt}$
    ($3'$)。
    \end{answer}
\end{question}

\begin{question}
    试求下列两电路的输入阻抗~$Z$~和导纳~$Y$。($10'$)
    \begin{figure}[H]
    \hfil
    \subfigure[]{\includegraphics{fig10}}\hfil
    \subfigure[]{\includegraphics{fig11}}
    \end{figure}
    \examvspace*{7cm}
    \begin{answer}
    (a) 容易求得~$Z=(1-\jimag 2)\unit{}{\ohm}$ ($3'$)，$Y=1/Z=(0.2+\jimag
    0.4)\unit{}{\siemens}$ ($2'$)；

    (b) $Z=(\jimag \omega L \dot{I}-r\dot{I})/\dot{I}=\jimag\omega L-r$
    ($3'$)，$Y=1/(\jimag\omega L-r)$ ($2'$)。
    \end{answer}
\end{question}

\begin{question}
    图示电路中，开关~S~闭合前电路已达稳定状态，$t=0$~时~S~闭合，求
    ~$t\geq 0$~时电容电压~$u_C$~的响应。($15'$)
    \begin{figure}[H]
    \hfill\includegraphics{fig06}
    \end{figure}
    \examvspace*{10cm}
    \begin{answer}
    先求~$u_C(0_-)$，解得~$u_C(0_-)=\unit{-7}{\volt}$ ($3'$)。则
    ~$u_C(0_+)=u_C(0_-)$，$u_\mathrm{OC}=\unit{3.5}{\volt}$ ($3'$)，利用外加
    电源法可求得~$R_\mathrm{eq}=\unit{4}{\ohm}$ ($3'$)，因此~$\tau
    =R_\mathrm{eq}C=\unit{2}{\second}$ ($3'$)。套用三要素法可得
    \[\begin{split}
    u_C(t)&=u_C(\infty)+[u_C(0_+)-u_C(\infty)]\mathrm{e}^{-t/\tau} \\
    &=(3.5-10.5\mathrm{e}^{-0.5t})\unit{}{\volt}\quad\quad (3')
    \end{split}\]
    \end{answer}
\end{question}
