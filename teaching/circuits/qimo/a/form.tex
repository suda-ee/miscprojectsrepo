\begin{question} % 01
    求图中各器件的功率(须说明是吸收还是发出)。($10'$)
    \begin{figure}[H]
    \hfill
    \input fig01
    \end{figure}
    \examvspace*{2cm}
    \begin{answer}
    $P_{\unit{2}{\ampere}} = 15 \times 2 = \unit{30}{\watt}$，发出。$P_{\unit{5}{\ohm}} = 15^2/5 = \unit{45}{\watt}$，吸收。$P_{\unit{15}{\volt}} = \unit{15}{\watt}$，发出。
    \end{answer}

\begin{question} % 02
    求图示电路中的 $I_1$ 和 $U_\mathrm{ab}$。($10'$)
    \begin{figure}[H]
    \hfill
    \input fig02
    \end{figure}
    \examvspace*{3cm}
    \begin{answer}
    $I = 10 /5 = \unit{2}{\ampere}$，$0.9I_1 = I$，$\therefore I_1 = I /
    0.9 = \unit{2.22}{\ampere}$，$\therefore U_\mathrm{ab} = 4 \times
      (I_1 - I) = \unit{0.889}{\volt}$。
    \end{answer}
\end{question}

\begin{question} %03
求图示电路中电压 $U$ 和电流 $I$。($10'$)
    %\examvspace{12pt}\\
    \begin{figure}[H]
    \hfill
    \input fig03
    \end{figure}
    \examvspace*{3cm}
    \begin{answer}
    列节点法方程。$(1/3 + 1/1)U = 10 - 2I$，$I = U / 3$。解得：$U = \unit{5}{\volt}$，$I = \unit{5/3}{\ampere}$。
    \end{answer}
\end{question}

\begin{question} % 04
    求图中的 $U_\mathrm{x}$。($10'$)
    \begin{figure}[H]
    \hfill
    \input fig04
    \end{figure}
    \examvspace*{3cm}
    \begin{answer}
    列节点电压方程，$(24 + 32)I - 24 \times 0.15 - 32 \times 0.05 U_\mathrm{x}
    = 20$，$U_\mathrm{x} = 32(I - 0.05 U_\mathrm{x}$。解得 $U_\mathrm{x} = \unit{8}{\volt}$。
    \end{answer}
\end{question}

\begin{question} % 05
    求图中的电压 $U_2$。($10'$)
    \begin{figure}[H]
    \hfill
    \input fig05
    \end{figure}
    \examvspace*{3cm}
    \begin{answer}
    列节点方程。$1/3 U_2 - 1/3 \times 2 = 3 - 2 I_1$，$I_1 = 2 /4$，解得 $U_2 = \unit{8}{\volt}$。
    \end{answer}
\end{question}

\begin{question} %06
    下图中负载电阻 $R_\mathrm{L}$ 可变，问 $R_\mathrm{L}$ 等于何值时可吸收最大功率，并求此功率。 ($10'$)
    \begin{figure}[H]
    \hfill
    \input fig06
    \end{figure}
    \examvspace*{6cm}
    \begin{answer}
    $(2 + 2)I_1 + 8 I_1 = 6$，$U_\mathrm{OC} = 2 I_1 + 2I_1 + 8 I_1 = \unit{6}{\volt}$，易解得 $I_\mathrm{SC} = \unit{3/2}{\ampere}$。$R_\mathrm{eq} = U_\mathrm{OC} / I_\mathrm{SC} = \unit{4}{\ohm}$。$P_\mathrm{max} = \frac{U_\mathrm{OC}^2}{4 R_\mathrm{eq}} = \unit{2.25}{\watt}$。
    \end{answer}
\end{question}

\begin{question} % 07

    \end{figure}
    \examvspace*{2cm}
    \begin{answer}

    Z = 1 + \frac{1}{1 / 2\jimag + 1 / (-\jimag)} = 1 -2 \jimag

    \end{answer}
\end{question}

\begin{question} %09
    下图所示电路，开关合在位置 1 时已达稳定状态，$t = 0$ 时开关由位置 1
    合向位置 2，求 $t \ge 0$ 时的电压 $u_L$。 ($15'$)
    \begin{figure}[H]
    \hfill
    \input fig09
    \end{figure}
    \examvspace*{10cm}
    \begin{answer}
    三要素法：$i_L = i_L(\infty) + [i_L(0_+) -
    i_L(\infty)]\mathrm{e}^{-t/\tau}$，$i_L(0_+) = i_L(0_-) =
    \unit{-4}{\ampere}$，$i_L(\infty) = \unit{1.2}{\ampere}$，$\tau =
    L/R_\mathrm{eq} = \unit{0.01}{\second}$，所以 $i_L(t) = 1.2 - 5.2
    \mathrm{e}^{-100t}$，$u_L(t) = L \frac{\mathrm{d}i_L}{\mathrm{d}t} =
    52 \mathrm{e}^{-100t},\, t\ge 0$。
    \end{answer}
\end{question}
