    \examvspace{12pt}\\
    \includegraphics{fig-1.pdf}

    \examvspace*{0cm}
    \begin{answer}

    \begin{align*}
    i_1+i_4-i_6&=0\\
    -i_2-i_4+i_5&=0\\

    \begin{answer}
    可写出~KVL~方程如下~(仅给出按顺时针的写法)：

\begin{shortanswer}[title={\begin{description}\item[二、]计算题~(每
题~10~分，共~50~分)\end{description}}]
\begin{question}
用网孔电流法求解图中的电流~$I_2$。
    \examvspace{12pt}\\
    \includegraphics{fig-5.pdf}

    \examvspace*{5cm}
    \begin{answer}
    设网孔电流都按顺时针方向，分别为~$I_\mathrm{m1}$、$I_\mathrm{m2}$。可列出网孔方程
    如下：
    \begin{align*}
    4I_\mathrm{m1}-I_\mathrm{m2}&=7\\
    -I_\mathrm{m1}+4I_\mathrm{m2}&=2
    \end{align*}
    (每式~3~分)~解得：
    $I_\mathrm{m1}=\unit{2}{\ampere},\,I_\mathrm{m2}=\unit{1}{\ampere}$。(2~分)~于是
    ~$I_2=I_\mathrm{m2}-I_\mathrm{m1}=\unit{-1}{\ampere}$。(2~分)
    \end{answer}
\end{question}
\begin{question}
    用结点电压法求图中~a、b~两点的电压。
    \examvspace{12pt}\\
    \includegraphics{fig-6.pdf}

    \examvspace*{5cm}
    \begin{answer}
    设~a、b~两点的电压分别是~$U_\mathrm{a}$、$U_\mathrm{b}$，可列结点方程如下：
    \begin{align*}
    \frac{4}{3}U_\mathrm{a}-U_\mathrm{b}&=2\\
    -U_\mathrm{a}+\frac{3}{2}U_\mathrm{b}&=0
    \end{align*}
    (每式~3~分)~解得~$U_\mathrm{a}=\unit{3}{\volt}$，$U_\mathrm{b}=\unit{2}{\volt}$。(每结
结果~2~分)
    \end{answer}
\end{question}
\begin{question}
    如下图所示电路，利用叠加定理求电压~$U$。
    \examvspace{12pt}\\
    \includegraphics{fig-7.pdf}

    \examvspace*{5cm}
    \begin{answer}
    电压源单独作用时，电流源置零相当于开路~(2~分)，可用串联分压得到
    ~$U^{(1)}=12\times 3/(3+9)=\unit{-4}{\volt}$(2~分)；电流源单独作用时，电压
    源置零相当于短路~(2~分)，可用串联分流得到：$U^{(2)}=3\times
    6/(6+3)\times 3=\unit{6}{\volt}$(2~分)。可得总的
    ~$U=U^{(1)}+U^{(2)}=\unit{2}{\volt}$。(2~分)
    \end{answer}
\end{question}
\begin{question}
    求图示电路的戴维宁或诺顿等效电路。
    \begin{figure}[H]
    \hfill
    \includegraphics{fig-8.pdf}
    \end{figure}

    \examvspace*{5cm}
    \begin{answer}
    利用简单的串并连等效容易求得等效内阻为~$R_{\rm eq}=\unit{2}{\ohm}$(3~分)；
    再利用叠加定理来求~$U_{\rm ab}$：电流源单独作用时求得
    ~$U_{\rm ab}^{(1)}=1\times 3/2\times 2/3=\unit{1}{\volt}$(3~分)；电压源
    单独作用时的~$U_{\rm ab}^{(2)}=-3\times
    4/(2+2+4)=\unit{-1.5}{\volt}$(3~分)；故开路电压
    ~$U_{\rm ab}=U_{\rm ab}^{(1)}+U_{\rm ab}^{(2)}=\unit{－0.5}{\volt}$(1~分)。
    \end{answer}
\end{question}
\begin{question}
    求图中电路在~ab~端口的戴维宁或诺顿等效电路。
    \examvspace{12pt}\\
    \includegraphics{fig-9.pdf}

    \examvspace*{5cm}
    \begin{answer}
    内部独立源全部置零后，成为一简单串联电路，容易求得等效内阻为：
    $R_{\rm eq}=\unit{10}{\ohm}$(5~分)；更容易解得开路电压~$u_{\rm
    OC}=10-1\times 5=\unit{5}{\volt}$(5~分)。
    \end{answer}
\end{question}
\end{shortanswer}

\begin{shortanswer}[title={\begin{description}\item[三、]计算分析题~(每
题~15~分，共~30~分)\end{description}}]
\begin{question}
    求下图电路中的电流~$I$。
    \begin{figure}[H]
    \hfill
    \includegraphics{fig-10.pdf}
    \end{figure}
    
    \examvspace*{12cm}
	\begin{answer}
	图略。当~\unit{5}{\ohm}~电阻开路时有方程
	\begin{gather*}
	I_1=I_{\rm x}+4 \,\,\,(3\mbox{~分})\\
	4I_{\rm x}+5I_{\rm x}+1\times I_1=12\,\,\,(3\mbox{~分})
	\end{gather*}
	解得~$I_{\rm x}=\unit{0.8}{\ampere}$，故~$U_{\rm OC}=3\times
	4+5I_{\rm x}=\unit{16}{\volt}$。(2~分)\\
	当~\unit{5}{\ohm}~电阻短路时有方程
	\begin{gather*}
	I_1=I_{\rm x}+4 \,\,\,(2\mbox{~分})\\
	4I_{\rm x}+5I_{\rm x}+1\times I_1=12\,\,\,(3\mbox{~分})
	\end{gather*}
	解得~$I_{\rm x}=\unit{0.8}{\ampere}$，故~$I_{\rm
	SC}=I_2+4=\unit{16/3}{\ampere}$。(1~分)\\
	可得~$R_{\rm eq}=\unit{3}{\ohm}$，$I=\unit{2}{\ampere}$。(1~分)
	\end{answer}
\end{question}
    \begin{question}
    下图所示电路中，已知~\unit{5}{\volt}~电压源~$U_{\rm S}$~支路电流
    ~$I_0=\unit{10}{\ampere}$，试确定受控电流源的控制系数~$\alpha$。
    \begin{figure}[H]
    \hfill
    \includegraphics{fig-11.pdf}
    \end{figure}

    \examvspace*{12cm}
	\begin{answer}
	用网孔法。可得右下角网孔的方程为
	\[
	(R_1+R_3)I_3-R_1I_1+R_3I_3=U_{\rm S}\,\,\,\,(6\mbox{~分})
	\]
	将其他各网孔电流代入得
	\[
	I_3=\frac{10-15\alpha}{4-3\alpha}\,\,\,\,(6\mbox{~分})
	\]
	又因为~$I_3=I_0=\unit{10}{\ampere}$，得~$\alpha=2$。(3~分)
	\end{answer}
    \end{question}
\end{shortanswer}
